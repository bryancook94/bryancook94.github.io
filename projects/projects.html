MySQL procedure defined 
                
DROP PROCEDURE IF EXISTS Rp_bc_core.proc_cd_sadler_buffet;
CREATE PROCEDURE Rp_bc_core.`proc_cd_sadler_buffet`()
begin

#insert into Rp_bc_core.bc_test(col1) values('3');
Drop table if exists Rp_bc_core.cd_sadler_buffet_temp1;
create table Rp_bc_core.cd_sadler_buffet_temp1
select distinct text,leftt from Rp_bc_core.cd_sadler_buffet
where text like '%chicken%' or text like '%lasagne%';

#insert into Rp_bc_core.bc_test(col1) values('4');
Drop table if exists Rp_bc_core.cd_sadler_buffet_temp2;
create table Rp_bc_core.cd_sadler_buffet_temp2
select distinct text,leftt from Rp_bc_core.cd_sadler_buffet
where text in ('"monday"','"tuesday"','"wednesday"','"thursday"');

#insert into Rp_bc_core.bc_test(col1) values('5');
Drop table if exists Rp_bc_reporting.rep_sadler_buffet;
create table Rp_bc_reporting.rep_sadler_buffet
select a.text,a.leftt,b.week,curdate() from (
select a.text,a.leftt,min(a.distance) mindistance from
(
select distinct a.text,a.leftt,b.text week,b.leftt weekpos,ABS(a.leftt-b.leftt) distance
from Rp_bc_core.cd_sadler_buffet_temp1 a,Rp_bc_core.cd_sadler_buffet_temp2 b
) a
group by a.text,a.leftt
) a
inner join (select distinct a.text,a.leftt,b.text week,b.leftt weekpos,ABS(a.leftt-b.leftt) distance from Rp_bc_core.cd_sadler_buffet_temp1 a,Rp_bc_core.cd_sadler_buffet_temp2 b) b on a.mindistance=b.distance ;


truncate Rp_bc_core.cd_sadler_buffet;
#insert into Rp_bc_core.bc_test(col1) values('6');
  end;
